"""
Ce module déploie l'application Flask pour la détection d'émotions.
Il sert d'interface entre le client web et la fonction d'analyse d'émotions.
"""

from flask import Flask, render_template, request
from EmotionDetection.emotion_detection import emotion_detector

# Initialisation de l'application Flask
app = Flask(__name__)

@app.route("/")
def render_index_page():
    """
    Cette fonction gère la route principale ('/')
    et affiche la page web index.html à l'utilisateur.
    """
    return render_template('index.html')

@app.route("/emotionDetector")
def emo_detector():
    """
    Cette fonction reçoit le texte depuis l'interface HTML,
    l'envoie à la fonction emotion_detector, et formate la réponse
    selon les exigences exactes du client.
    """
    # 1. On récupère le texte tapé par l'utilisateur dans l'interface
    text_to_analyze = request.args.get('textToAnalyze')

    # 2. On passe ce texte à notre fonction d'analyse
    response = emotion_detector(text_to_analyze)

    # 3. On formate la chaîne de caractères EXACTEMENT comme demandé par le client
    formated_response = (
        f"Pour l'énoncé donné, la réponse du système est "
        f"'anger': {response['anger']}, "
        f"'disgust': {response['disgust']}, "
        f"'fear': {response['fear']}, "
        f"'joy': {response['joy']} et "
        f"'sadness': {response['sadness']}. "
        f"L'émotion dominante est {response['dominant_emotion']}."
    )

    return formated_response

if __name__ == "__main__":
    # On lance l'application sur le port 5000
    app.run(host="0.0.0.0", port=5000)
